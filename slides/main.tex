\section{Introduction}

\begin{frame}{Project Goal}
    \begin{itemize}
        \item The project compares a classical convolutional model and a transformer-style model under the same dataset setting.
        \item The two models are trained, saved, reloaded, and evaluated in the same notebook workflow.
        \item The main objective is to analyze not only accuracy, but also training behavior, confidence, and difficult examples.
    \end{itemize}
\end{frame}

\begin{frame}{Why Compare AlexNet and DHVT?}
    \begin{table}
        \centering
        \begin{tabular}{p{3.2cm}p{6.9cm}}
            \toprule
            Model & Role in the study \\
            \midrule
            AlexNet & CNN baseline with strong image inductive bias and a clear layer-by-layer structure. \\
            DHVT & Transformer-style model adapted to small datasets through overlapped patch embedding and DAFF blocks. \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}
    \begin{itemize}
        \item This comparison contrasts local convolutional processing with token-based attention.
        \item Both models use the same dataset configuration to keep the experiment fair.
    \end{itemize}
\end{frame}

\section{Models}

\begin{frame}{AlexNet}
    \begin{itemize}
        \item AlexNet uses stacked convolutions, ReLU, pooling, and fully connected layers.
        \item In the notebook, it serves as the CNN baseline for the comparison.
        \item The convolution operation is written as:
    \end{itemize}

    \[
    Y_{i,j}=\sum_u\sum_v K_{u,v}X_{i+u,j+v}
    \]

    \[
    \mathrm{ReLU}(x)=\max(0,x)
    \]

    \begin{itemize}
        \item This structure emphasizes local feature extraction and spatial hierarchy.
    \end{itemize}
\end{frame}

\begin{frame}{DHVT}
    \begin{itemize}
        \item DHVT is a transformer-style architecture adapted to smaller datasets.
        \item The model includes SOPE, class token, positional embeddings, attention blocks, and DAFF.
        \item Patch-token embedding can be summarized as:
    \end{itemize}

    \[
    X \in \mathbb{R}^{B \times C \times H \times W}
    \rightarrow
    E \in \mathbb{R}^{B \times N \times d}
    \]

    \[
    Z_0=[x_{\mathrm{cls}};E], \qquad Z=Z_0+P
    \]

    \begin{itemize}
        \item The model is trained from scratch rather than used only for fine-tuning.
    \end{itemize}
\end{frame}

\begin{frame}{Attention and DHVT Block}
    \[
    Q=ZW_Q,\qquad K=ZW_K,\qquad V=ZW_V
    \]

    \[
    \mathrm{Attention}(Q,K,V)=\mathrm{softmax}\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
    \]

    \[
    Z' = Z + \mathrm{MHSA}(\mathrm{LN}(Z))
    \]

    \[
    Z'' = Z' + \mathrm{DAFF}(\mathrm{LN}(Z'))
    \]

    \begin{itemize}
        \item The DAFF block adds a convolution-aware feed-forward stage.
        \item This design helps the transformer preserve useful locality for small datasets.
    \end{itemize}
\end{frame}

\section{Training and Difficulties}

\begin{frame}{Training Strategy}
    \begin{itemize}
        \item Both models are trained under the same dataset configuration.
        \item Checkpoints are saved and reloaded inside the notebook.
        \item The objective function is the cross-entropy loss:
    \end{itemize}

    \[
    \mathcal{L} = - \sum_{c=1}^{C} y_c \log(\hat{p}_c)
    \]

    \begin{itemize}
        \item Validation accuracy is monitored to keep the best weights for each model.
    \end{itemize}
\end{frame}

\begin{frame}{Main Difficulties and Solutions}
    \begin{table}
        \centering
        \begin{tabular}{p{4cm}p{5.8cm}}
            \toprule
            Difficulty & Solution \\
            \midrule
            Fair comparison & Use one notebook and one shared dataset configuration. \\
            Small-dataset transformer training & Use a DHVT-style architecture trained from scratch. \\
            Homework-style presentation & Keep the workflow step-by-step in the notebook. \\
            Reusability & Support multiple datasets and custom folders. \\
            Long training runs & Print intermediate progress during training. \\
            \bottomrule
        \end{tabular}
    \end{table}
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item DHVT is the stronger model in the current experiment.
        \item It achieves better validation accuracy while using fewer parameters than AlexNet.
        \item The notebook now supports training, saving, reloading, inference, and visual comparison in one workflow.
        \item The project also shows that model evaluation should include curves, confidence behavior, and qualitative error analysis.
    \end{itemize}
\end{frame}
